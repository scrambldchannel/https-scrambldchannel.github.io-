    try
        @eval using Revise
        @async Revise.wait_steal_repl_backend()
    catch e
        @warn(e.msg)
    end

Simple module overview¶














SWAPI/src/SWAPI.jl¶
module SWAPI

    export request_data

    include("request.jl")

end
















SWAPI/src/request.jl¶
using HTTP
using JSON

function request_data(url)
    response = HTTP.get(url)
    return JSON.print(JSON.parse(response), 4)
end
